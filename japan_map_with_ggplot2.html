pref<-as.character(unique(jpn$NAME_1))
pref_test <- pref[42]
jpn.sub <- subset(jpn,NAME_1==pref_test)
map <- fortify(jpn.sub)

## Regions defined for each Polygons

map_id <- unique(map$id)
dd <- data.frame(
  id   = map_id,
  name = subset(jpn,ID_2 %in% map_id)$NAME_2,
  pref_id = pref_test
)
